curl -G -X GET http://10.3.225.203:1026/v2/entities?type=Daily_COVID_Cases_In_City&options=keyValues

curl -i -G -X GET "http://10.3.225.203:1026/v2/entities" -d "type=Daily_COVID_Cases_In_City" -d "options=count" -d "limit=1"
